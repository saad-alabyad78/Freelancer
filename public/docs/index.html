Introduction

Base URL: http://localhost

This documentation aims to provide all the information you need to work with our API.

As you scroll, you'll see code examples for working with the API in different programming languages in the dark area to the right (or as part of the content on mobile). You can switch the language used with the tabs at the top right (or from the nav menu at the top left on mobile).

Authenticating requests

To authenticate requests, include an Authorization header with the value "Bearer {YOUR_AUTH_KEY}".

All authenticated endpoints are marked with a requires authentication badge in the documentation below.

You can retrieve your token by visiting your dashboard and clicking Generate API token.

Auth Managment

APIs to manage the login and logout

POST api/login

requires authentication

Example request:
curl --request POST \
    "http://localhost/api/login" \
    --header "Authorization: Bearer {YOUR_AUTH_KEY}" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"email\": \"jordane58@example.net\",
    \"password\": \"~xls!f%Z9,s^cm#_O\"
}"

Create/Store New Company .

requires authentication

Example request:
curl --request POST \
    "http://localhost/api/company/store" \
    --header "Authorization: Bearer {YOUR_AUTH_KEY}" \
    --header "Content-Type: multipart/form-data" \
    --header "Accept: application/json" \
    --form "industry_name=autem"\
    --form "name=lpd"\
    --form "description=Recusandae ex explicabo voluptates deserunt rerum repellat."\
    --form "size=o"\
    --form "city=دمشق"\
    --form "region=pnspgkmlpjzeg"\
    --form "street_address=jijjihsinevnkl"\
    --form "contact_links[]=zloxnooseugsdbonbckhuyvonrvmbqsllppjezupjiy"\
    --form "company_phones[]=molestiae"\
    --form "profile_image=@/tmp/phppJU49d" \
    --form "background_image=@/tmp/phpxxzjYm" \
    --form "gallery_images[]=@/tmp/phpEiUNPV" 
const url = new URL(
    "http://localhost/api/company/store"
);

const headers = {
    "Authorization": "Bearer {YOUR_AUTH_KEY}",
    "Content-Type": "multipart/form-data",
    "Accept": "application/json",
};

const body = new FormData();
body.append('industry_name', 'autem');
body.append('name', 'lpd');
body.append('description', 'Recusandae ex explicabo voluptates deserunt rerum repellat.');
body.append('size', 'o');
body.append('city', 'دمشق');
body.append('region', 'pnspgkmlpjzeg');
body.append('street_address', 'jijjihsinevnkl');
body.append('contact_links[]', 'zloxnooseugsdbonbckhuyvonrvmbqsllppjezupjiy');
body.append('company_phones[]', 'molestiae');
body.append('profile_image', document.querySelector('input[name="profile_image"]').files[0]);
body.append('background_image', document.querySelector('input[name="background_image"]').files[0]);
body.append('gallery_images[]', document.querySelector('input[name="gallery_images[]"]').files[0]);

fetch(url, {
    method: "POST",
    headers,
    body,
}).then(response => response.json());
